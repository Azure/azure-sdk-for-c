cmake_minimum_required (VERSION 3.10)

project (az_iot_common_test LANGUAGES C)

set(CMAKE_C_STANDARD 99)

include(AddTestCMocka)

# Generate map file
if(MSVC)
    add_link_options(/MAP)
elseif(CMAKE_C_COMPILER_ID MATCHES "Clang")
    add_link_options(-Wl,-map,az_iot_common_test.map)
    add_link_options(-Os)
else()
    add_link_options(-Xlinker -Map=az_iot_common_test.map)
    add_link_options(-Os)
endif()

add_cmocka_test(az_iot_common_test SOURCES
                main.c
                test_az_iot_common.c
                COMPILE_OPTIONS ${DEFAULT_C_COMPILE_FLAGS}
                LINK_TARGETS                        
                    az_iot_common
                    az_core
                    ${PAL}                
                )
